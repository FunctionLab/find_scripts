1function [nodeBel,edgeBel] = UGM_Infer_ConditionalC(clampedNB,clampedEB,nodePot,edgePot,edgeMap,edgeEnds,clamped)
 2
 3[nNodes,~] = size(nodePot);
 4nEdges = size(edgePot,3);
 5
 6
 7nodeBel = zeros(size(nodePot));
 8clampVar = 1;
 9regulVar = 1;
10regulVarMap = zeros(size(clamped));
11for n = 1:nNodes
12   if clamped(n)~=0
13       nodeBel(n,clamped(n)) = 1;
14       clampVar = clampVar+1;
15   else
16       nodeBel(n,:) = clampedNB(regulVar,:);
17       regulVarMap(n) = regulVar;
18       regulVar = regulVar + 1;
19   end
20end
21
22% Construct edge beliefs
23edgeBel = zeros(size(edgePot));
24for e = 1:nEdges
25   n1 = edgeEnds(e,1);
26   n2 = edgeEnds(e,2);
27   
28   if clamped(n1)~=0
29       if clamped(n2)~=0
30           % This edge is between clamped variables
31           edgeBel(clamped(n1),clamped(n2),e) = 1;
32       else
33           % n1 is a clamped variable, n2 is a regular variable
34           edgeBel(clamped(n1),:,e) = clampedNB(regulVarMap(n2),:);
35       end
36   else
37       if clamped(n2)~=0
38           % n2 is a clamped variable, n1 is a regular variable
39           edgeBel(:,clamped(n2),e) = clampedNB(regulVarMap(n1),:)';
40       else
41           % This edge was present in the clamped graph
42           edgeBel(:,:,e) = clampedEB(:,:,edgeMap(e));
43       end
44   end
45end
